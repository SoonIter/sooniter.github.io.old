```javascript
/**
 * @param {ListNode} head
 * @return {ListNode}
 */
var reverseList = function(head) {
    if(head==null){
        return null;
    }
    let prev = head;
    let next = head.next;
    while(next!=null){
        let temp = next.next;
        next.next = prev;
        if(prev==head){
            prev.next=null
        }
        prev = next;
        next = temp;
    }
    return prev;
};
```
