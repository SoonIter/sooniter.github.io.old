## dijskra算法

主要用于求到各点的距离





1.代码

```cpp
class Solution
{
    int n; //n个结点
    int s; //起点
    int array[100][100];
    bool visited[100];
    int distance[100];
    int parent[100];

public:
    void input() //调整用户输入，转化矩阵格式
    {
        cin >> n;
        cin >> s;

        for (int i = 0; i < n; i++)
        {
            for (int j = 0; j < n; j++)
            {
                cin >> array[i][j]; //输入矩阵
            }
        }
    }
    void init() //初始化
    {
        for (int i = 0; i < n; i++)
        {
            visited[i] = false;
            distance[i] = infinity;
            parent[i] = -1;
        }
    }
    void update(int preAdd)
    {
        int i = preAdd;
        for (int j = 0; j < n; j++)
        {
            int weight = array[i][j];
            if (weight != 0 && visited[j] == false)
                distance[j] = weight + distance[i];
        }
    }
    int scan() //找出未访问中的最小值所在位置
    {
        int pos = -1;
        int min = infinity;
        for (int i = 0; i < n; i++)
        {
            if (visited[i] == false && distance[i] < min)
            {
                min = distance[i];
                pos = i;
            }
        }
        return pos;
    }
    void add(int i, int parent)
    {
        if (i < n && i >= 0)
        {
            visited[i] = true;
            this->parent[i] = parent;
        }
    }

    void dijskra()
    { //dijskra最短路径算法
        init();
        int preAdd = s - 1; //起点
        distance[s - 1] = 0;
        visited[s - 1] = true;
        for (int i = 1; i < n; i++)
        {
            int parent = preAdd; //暂存一下父亲
            update(preAdd);
            preAdd = scan();
            add(preAdd, parent);
        }
    }

    void print()
    {
        for (int i = 0; i < n; i++)
        {
            cout << s << "到" << i + 1 << "点的距离是" << distance[i] << endl;
        }
    }
};
```
